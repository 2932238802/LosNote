3. 

`std::all_of`
- 函数格式 
  - 
  ```c++
  template<class InputIt, class UnaryPredicate>
  bool all_of(InputIt first, InputIt last, UnaryPredicate p);
  ```
  UnaryPredicate 就是 谓词
  
  比如:
  ```c++
      int main() {
      vector<int> arr{2, 4, 6, 10, 8};
      auto res =
          all_of(arr.begin(), arr.end(), [](const int &a) { return a % 2 == 0; });
      cout << res << endl; // 1
      auto res2 =
          all_of(arr.begin(), arr.end(), [](const int &a) { return a < 9; });
      cout << res2 << endl; // 0
      return 0;
    } 
  ```
