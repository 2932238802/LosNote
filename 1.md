vector:
```c++
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
int main() {
  auto func = [](const std::vector<int> &arr) {
    for (auto a : arr) {
      cout << a << " ";
    }
    cout << endl;
  };
  std::vector<int> vec{1, 2, 3, 4, 5, 6};
  std::for_each(vec.begin(), vec.begin() + 3, [](int &a) { a += 1; });
  // 2 3 4 4 5 6
  // 左闭右开
  func(vec);
  return 0;
}
```

可以实现对数组进行相加 现在尝试字符串
```c++
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

int main() {
  auto func = [](const std::string &s) {
    for (auto a : s) {
      cout << a << " ";
    }
    cout << endl;
  };
  std::string s{"abcdef"};

  std::for_each(s.begin(), s.begin() + 3, [](char &a) { a += 1; });
  // b c d d e f
  // 左闭右开

  func(s);
  return 0;
}
```
