```rust
struct Stu {
    height: u32,

    // 人家 传过来是 years 把这个 改成old
    // #[serde(rename = "years")]
    #[serde(default = "default_limit")]
    old: u32,
}

fn default_limit() -> u32 {
    10
}

fn main() {
    let json_origin = r#"
    {
        "height" :2
    }
    "#;

    let user_struct: Stu = serde_json::from_str(&json_origin).unwrap();

    println!("{}", &user_struct.height);
    println!("{}", &user_struct.old);
}
```

- 通过这个方式 可以 传递 默认的值
