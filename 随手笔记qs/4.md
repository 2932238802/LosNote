5. 

```rust
use serde::{Deserialize, Serialize};

#[derive(Serialize, Deserialize, Debug)]
struct Stu {
    // 人家 传过来是 years 把这个 改成old
    #[serde(rename = "years")]
    old: u32,

    height: u32,
}

fn main() {
    let json_origin = r#"
    {
        "years" : 1,
        "height" :2
    }
    "#;

    let user_struct: Stu = serde_json::from_str(&json_origin).unwrap();

    println!("{}", &user_struct.height);
    println!("{}", &user_struct.old);
}

```



- 这样是可以的
