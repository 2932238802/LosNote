2. [c++]

std::format

```c++
int main()
{
  std::cout << std::format("这里有 {} 个 西瓜", 22) << std::endl;
  // 这里有 22 个 西瓜

  int val = 10;
  std::cout << std::format("3 + 4 = {}", val - 3) << std::endl;
  // 3 + 4 = 7

  std::string name = "xiao ming";
  int score = 20;
  std::cout << std::format("{1} 获得了 {0} 分, {1} 是极棒的!", score, name)
            << std::endl;
  // xiao ming 获得了 20 分, xiao ming 是极棒的!
  // 默认 索引从0开始

  std::cout << std::format("|{:>10}|", "hello") << std::endl;
  // |     hello|
  // 右对齐

  double pi = 3.1415926;
  std::cout << std::format("{:.2f}", pi) << std::endl;
  // 3.14

  // {:[填充][对齐][宽度].[精度][类型]}

  int num = 255;
  std::cout << std::format("十六进制: {:x}", num) << std::endl;
  std::cout << std::format("十六禁止: {:X}", num) << std::endl;
  std::cout << std::format("八进制: {:o}", num) << std::endl;
  std::cout << std::format("十六进制&前缀: {:#x}", num) << std::endl;
  std::cout << std::format("十六进制&前缀: {:#X}", num) << std::endl;
  std::cout << std::format("二进制: {:b}", num) << std::endl;
  // 十六进制: ff
  // 十六禁止: FF
  // 八进制: 377
  // 十六进制&前缀: 0xff
  // 十六进制&前缀: 0XFF
  // 二进制: 11111111

  return 0;
}
```



3. [c++]

```c++
int main()
{
  std::cout << std::numeric_limits<int>::max() << std::endl;
  std::cout << std::numeric_limits<int>::min() << std::endl;
  std::cout << std::numeric_limits<int>::lowest() << std::endl;

  std::cout << std::numeric_limits<double>::max() << std::endl;
  std::cout << std::numeric_limits<double>::min() << std::endl;
  std::cout << std::numeric_limits<double>::lowest() << std::endl;
  // 2147483647
  // -2147483648
  // -2147483648
  // 1.79769e+308
  // 2.22507e-308
  // -1.79769e+308

  return 0;
}
```

5. [c++]

```c++
int main()
{
  int i = 10;
  std::strong_ordering s{i <=> 9};

  if (s == std::strong_ordering::less)
    std::cout << "小于";
  else if (s == std::strong_ordering::greater)
    std::cout << "大于";
  else
    std::cout << "等于";
  // 大于

  if (std::is_lt(s))
  {
    std::cout << "小于";
  }
  else if (std::is_gt(s))
  {
    std::cout << "大于";
  }
  else
  {
    std::cout << "等于";
  }

  // 大于

  return 0;
}
```
