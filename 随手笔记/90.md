```rust
use std::collections::HashMap;
fn main(){

    let teams = vec![String::from("a"),String::from("b"),String::from("c")];

    let numbers = vec![1,2];

    let team_to_number:HashMap<_,_> = teams.iter().zip(numbers.iter()).collect();

    println!("{}",team_to_number[&teams[0]]);

    let get_number = team_to_number.get(&teams[2]);

    match get_number{
        Some(number) =>{
            println!("哈希表里面c对应的数字是: {}",number);
        }
        None =>{
            println!("无多少数据")
        }
    }

}
```
