3. [c++]

std::array 是一个 固定大小的容器

提供了固定大小的数组功能

```c++
int main() {
     std::array<int, 4> contain{};

     // []
     cout << contain[0] << std::endl; // 0

     contain = {1,2,3,4};

     cout << contain[1] << std::endl; // 2

     contain.fill(3);
     
     cout << contain[3] << std::endl; // 3
     
     return 0; 
}
```
