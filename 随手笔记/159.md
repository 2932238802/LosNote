```

- ```c++
  requires (参数列表) {
      需求1;
      需求2;
      ...
  }
  ```

- 对的  requires 花括号里 可以写 简单需求 也可以写 复合需求

- ```c++
  requires(T t) {
      t.make_sound(); // 只要有这个函数就行，不管返回什么
      t + t;          // 只要支持加法就行
  }
  ```

- ```c++
  requires(T t) {
      typename T::value_type; // 检查 T 里面有没有定义 value_type
  }

- ```c++
  requires(T t) {
      t.make_sound();
      requires std::copyable<T>; // 除了能叫，T 还得是可拷贝的
  }

```c++


template <typename T>
concept Animal = requires (T t) {
    t.make_sound();
};

template <typename T>
concept Vege = requires (T t) {
    t.make_sun();
};

template <typename T>
concept Combine = Animal<T> && Vege<T>; 

class Cat{
    public:
    void make_sound() const{
        cout << "miew miew miew" << endl;
    };
};

class Dog{
    public:
    void make_sound() const{
        cout << "wang wang wang" << endl;
    };
};

class Scallion{
    public:
    void make_sun() const
    {
        cout << "111" << endl;
    }
};

class Garlic{
    public:
    void make_sun() const
    {
        cout << "222" << endl;
    }
};

class Other{
    public:
    void make_sun() const{
        cout << "hahahha";
    }
    void make_sound() const{
        cout << "dududu";
    }
};



class Market{
public:
    void process_item(const Animal auto& a)
    {
        a.make_sound();
    }
    void process_item(const Vege auto& b)
    {
        b.make_sun();
    }
    void process_item(const Combine auto & c) 
    {
        cout << "----" << endl;
        c.make_sun();
        c.make_sound();
        cout << "----" << endl;
    }
private:
};


int main()
{
    Dog d;
    Garlic g;
    Cat c;
    Scallion s;
    Other o;

    Market m;
    m.process_item(d);
    m.process_item(g);
    m.process_item(c);
    m.process_item(s);
    m.process_item(o);

    return 0;
}
