```C++
bool RecBoxItem::eventFilter(QObject *watched, QEvent *event)
{
    if(watched == ui->recMusicBtn)
    {
        if(QEvent::Enter == event->type())
        {
            ui->recMusicBtn->setCursor(Qt::PointingHandCursor);
            animation = new QPropertyAnimation(ui->imageBox,"geometry");
            animation->setDuration(400);
            animation->setStartValue();
        }else if(QEvent::Leave == event->type())
        {
            ui->recMusicBtn->unsetCursor();
        }
        return true;
    }
    return QWidget::eventFilter(watched,event);
}
```

4. Qt 生成随机字符串的方式？
```c++
QString generateRandomString(int length)
{
    // 定义字符池：包含所有你希望出现在随机字符串中的字符
    const QString possibleCharacters("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789");
    const int possibleCharacterCount = possibleCharacters.length();

    QString randomString;
    randomString.reserve(length);
    for (int i = 0; i < length; ++i)
    {
        int randomIndex = QRandomGenerator::global()->bounded(possibleCharacterCount);
        QChar nextChar = possibleCharacters.at(randomIndex);
        randomString.append(nextChar);
    }
    return randomString;
}

```
