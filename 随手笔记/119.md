4. [rust]

use serde::{Serialize};
use axum::http::StatusCode;

/// 翻译返回
/// code 500 
#[derive(Serialize)]
pub struct TranslationResponse {
    pub code: u16,
    pub result: String,
    pub data: Option<String>, 
}


impl TranslationResponse {

    /// success
    /// res 就是 api 翻译的结果
    pub fn success(res : String) -> Self {
        TranslationResponse { code: 200 , result: "翻译成功!".to_string(), data: Some(res) }
    }
    
    pub fn error(res : String) -> Self {
        TranslationResponse { code: 500 , result: "翻译失败!".to_string(), data: Some(res) }
    }
}

这个 200 和 500 如何 用 StatusCode 封装呢？



- StatusCode::OK
- StatusCode::INTERNAL_SERVER_ERROR
